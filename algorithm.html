1.1 Bubble Sort

1.1.1 basic

void BubbleSort1(int a[], int n) {
    int i, j;
    for ( i=0; i<n; i++ ) {
        for ( j=0; j<n-i; j++ ) {
            if ( a[j-1]>a[j] )
                swap(a[j-1], a[j]);
        }
    }
}

1.2 Insertion Sort

1.2.1 basic

void InsertSort1(int a[], int n) {
    int i, j;
    for ( i=1; i<n; i++ ) {
        for ( j=i-1; j>=0; j-- )
            if ( a[j] < a[i] )
                break;          // this is what a[i] should be

        if ( j != i-1) {        // insert it into a[j+1]
            int tmp = a[i];
            for ( int k=i-1; k>j; k++ )
                a[k+1] = a[k];
            a[k+1] = tmp;
        }
    }
}

1.2.2 better

void InsertSort2(int a[], int n) {
    int i, j;
    for ( i=1; i<n; i++ ) {
        if ( a[i] < a[i-1] ) {
            int tmp = a[i];
            for ( j=i-1; j>=0 && a[j] > tmp; j-- )
                a[j+1] = a[j];
            a[j+1] = tmp;
        }
    }
}

1.2.3 betterer

void InsertSort3(int a[], int n) {

        for ( j=i-1; j>=0 && a[j] > a[j+1]; j-- )
            swap(a[j], a[j+1]);
}

1.3 Shell Sort

1.3.1 basic

void shellsort1(int a[], int n) {
    int i, j, gap;

    for ( gap=n/2; gap>0; gap/=2 ) {
        for ( i=0; i<gap; i++ ) {
            for ( j=i+gap; j<n; j+=gap ) {
                if ( a[j] < a[j-gap] ) {
                    int tmp = a[j];
                    int k = j-gap;
                    while ( k>=0 && a[k]>tmp ) { // do swap
                        a[k+gap] = a[k];
                        k -= gap;
                    }
                    a[k+gap] = tmp;
                }
            }
        }
    }
}

1.3.2 better

void shellsort2(int a[], int n) {

1.4 Select Sort

void selectsort(int a[], int n) {
    int i, j, nMinIndex;
    for ( i=0; i<n; i++ ) {
        nMinIndex = i;
        for ( j=i+1; j<n; j++ ) {
            if ( a[j]<a[nMinIndex] ) {
                nMinIndex = j;
            }
        }
        swap(a[j], a[nMinIndex]);
    }
}

1.6 Quick Sort

1.6.1 Adjust Array

int AdjustArray(int s[], int l, int r)
{
  int i = l, j = r;
  int x = s[i];
  while ( i < j ) {
    while ( i < j && s[j] >= x )
      j--;
    if ( i < j ) {
      s[i] = s[j];
      i++;
    }

    while ( i < j && s[i] <= x)
      i++;
    if ( i < j ) {
      s[j] = s[i];
      j--;
    }
  }
  s[i] = x;

  return i;
}

1.6.2 Divide and Conquer

void dac(int s[], int l, int r)
{
  if ( l < r ) {
    int i = AdjustArray(s, l, r);
    dac(s, l, i-1);
    dac(s, i+1, r);
  }
}

1.6.3 basic

void quick_sort(int s[], int l, int r)
{
  if ( l < r ) {
    int i = l, j = r, x = s[i];
    while ( i < j ) {
      while ( i < j && s[j] >= x )
        j--;
      if ( i < j )
        s[i++] = s[j];

      while ( i < j && s[i] <= x )
        i++;
      if ( i < j )
        s[j--] = s[i];
    }
    s[i] = x;
    quick_sort(s, l, i-1);
    quick_sort(s, i+1, r);
  }
}

1.7 Heap Sort

1.7.1 Fixup (make the tree follow the definition)

void minHeapFixUp( int a[], int i )
{
  int j, tmp;

  tmp = a[i];
  j = (i-1)/2;
  while ( j>=0 && i!=0 )
    {
      if ( a[j]<=tmp )
        break;
      a[i] = a[j];
      i = j;
      j = (i-1)/2;
    }
  a[i] = tmp;
}

void minHeapFixDown( int a[], int i, int n )
{
  int j, tmp;

  tmp = a[i];
  j = 2*i + 1;
  while ( j<n )
    {
      if ( j+1<n && a[j+1]<a[j] )
        {
          j++;
        }
      if ( a[j]>=tmp )
          break;
      a[i] = a[j];
      i = j;
      j = 2*i + 1;
    }
  a[i] = tmp;
}

1.8 Radix Sort

int maxbit(int data[], int n)
{
  int maxData = data[0];
  for ( int i=0; i<n; i++ )
    {
      if ( maxData<data[i] )
        {
          maxData = data[i];
        }
    }
  int d = 1;
  int p = 10;
  while ( maxData>=p )
    {
      p *= 10;
      d++;
    }
  return d;
}

void radixSort(int data[], int n)
{
  int d = maxbit(data, n);
  int *tmp = new int[n];
  int *count = new int[10];
  int i, j, k;
  int radix = 1;
  for( i=1; i<=d; i++ )
    {
      for( j=0; j<10; j++ )
        {
          count[j] = 0;
        }
      for( j=0; j<n; j++ )
        {
          k = (data[j]/radix) % 10;
          count[k]++;
        }
      for( j=1; j<10; j++ )
        {
          count[j] = count[j-1] + count[j];
        }
      for( j=n-1; j>=0; j-- )
        {
          k = (data[j]/radix) % 10;
          tmp[count[k]-1] = data[j];
          count[k]--;
        }
      for( j=0; j<n; j++ )
        {
          data[j] = tmp[j];
        }
      radix = radix * 10;
    }
  delete []tmp;
  delete []count;
}

1.9 Find repeat number

// Idea: if a number exists, it would be marked by a flag
// So, those number with flag are thought repeated
int Repeat(int *array, int n)
{
  for( int i=0; i<n; i++ ) {
    index = a[i]>=n ? a[i]-n : a[i];
    if( index>=n ) {
        return array[i];
    }
    else {
      array[ index ] += n;
    }
}

1.11 Find one single number(with others are trippled or more)

void findSingle(int a[], int n, int repeat)
{
  int count[sizeof(int)*8];
  for( int i=0; i<sizeof(int)*8; i++ ) {
    count[i] = 0;
    for( int j=0; j<n; j++ ) {
      count[i] += ((a[j]>>i) & 1);
    }
  }
  int result = 0;
  for( int i=0; i<sizeof(int)*8; i++ ) {
    if( count[i]%repeat != 0 ) {
      result += (1<<i);
    }
  }
  return result;
}

// this function generates n random number
// to make them
void getRandomNum(int a[], int n, int start, int end)
{
  std::set<int> set_a;
  srand(time(NULL));
  for( int i=end-n; i<end; i++ ) {
    int num = (rand() % i) + start;
    if( set_a.find(num) == set_a.end() ) {
      set_a.insert(num);
    }
    else {
      set_a.insert(i);
    }
  }
  i = 0;
  std::set<int>::iterator pos;
  for( pos=set_a.begin(); pos!=set_a.end(); pos++ ) {
    a[i++] = *pos;
  }
}

void generateNum(int sum, int count) {
  int a[count], result[count];
  getRandomNum(a, count-1, 0, sum);
  sort(a, a+count-1);
  a[count] = sum;
  for( int i=0; i<count; i++ ) {
    result[i] = a[i] - a[i-1];
  }
}

1.13 Find a number in an array, the difference of two neighbor in this array is 1

void findNum(int a[], int n, int y)
{
  int index = abs(y-a[0]);
  while( index<n ) {
    if( a[index] == y ) {
      return index;
    }
    else {
      index += abs(y-index);
    }
  }
  return -1;
}

2 Document Distance

Angle of two document vector, each dimention is a word so the distance of two document of the inner product of two document:

\begin{split} D[w]&=\#\ of\ w\ in\ D\\ d'(D_{1},D_{2})&=D_{1} \cdot D_{2}\\ &=\sum_{w}{D_{1}[w] \cdot D_{2}[w]}\\ d''(D_{1},\ D_{2})&=\frac{d'(D_{1},\ D_{2})}{||D_{1}|| \cdot ||D_{2}||}\\ d(D_{1},\ D_{2})&=\arccos{d''(D_{1},\ D_{2})} \end{split}
